// 6. 長官狀態回報功能 function reportStatus(person, action) { const now = Date.now(); const actionKey = action === '蒞轄' ? '蒞' : '離'; // 寫入 Firebase db.ref(`officials/${person}/${actionKey}`).set({ timestamp: now, status: action }); } // 監聽長官狀態更新 UI db.ref('officials').on('value', (snapshot) => { const data = snapshot.val(); if (data) { Object.keys(data).forEach(person => { // 更新蒞轄時間 if (data[person].蒞) { const timeStr = new Date(data[person].蒞.timestamp).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}); const el = document.getElementById(`time-${person}-蒞`); if (el) el.innerText = timeStr; } // 更新離轄時間 if (data[person].離) { const timeStr = new Date(data[person].離.timestamp).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}); const el = document.getElementById(`time-${person}-離`); if (el) el.innerText = timeStr; } }); } });
